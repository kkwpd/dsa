#include <iostream>
using namespace std;
#define V 6
#define INF 9999

int g[V][V] = {
    {0,10,0,0,15,5},
    {10,0,10,30,0,0},
    {0,10,0,12,5,0},
    {0,30,12,0,0,20},
    {15,0,5,0,0,0},
    {5,0,0,20,0,0}
};

int main() {
    int d[V], p[V] = {-1};
    bool v[V] = {0};
    string n[] = {"City Center","Mall","Railway","Airport","Stadium","Park"};

    for(int i=0;i<V;i++) d[i]=INF;
    d[0]=0;

    for(int i=0;i<V;i++) {
        int u=-1;
        for(int j=0;j<V;j++) if(!v[j] && (u==-1 || d[j]<d[u])) u=j;
        v[u]=1;

        for(int x=0;x<V;x++)
            if(g[u][x] && d[u]+g[u][x]<d[x])
                d[x]=d[u]+g[u][x], p[x]=u;
    }

    cout<<"From City Center:\nNo  Place          Distance  Path\n";
    for(int i=0;i<V;i++) {
        cout<<i<<"   "<<n[i];
        for(int j=15;j>n[i].size();j--) cout<<" ";
        cout<<d[i]<<" km    ";
        for(int x=i;x>=0;x=p[x]) { cout<<x<<(p[x]>=0?"<-":""); }
        cout<<endl;
    }
}
