#include <iostream>
using namespace std;
#define V 6
#define INF 9999

int graph[V][V] = {
    {0,   100, INF, 200, 150, 300},
    {100,   0, 120, INF, INF, 180},
    {INF, 120,   0, 250, INF, INF},
    {200, INF, 250,   0, 180, INF},
    {150, INF, INF, 180,   0, 100},
    {300, 180, INF, INF, 100,   0}
};

int main() {
    int key[V], parent[V];
    bool inMST[V] = {false};
    
    for(int i=0; i<V; i++) key[i] = INF;
    key[0] = 0; parent[0] = -1;

    for(int i=0; i<V; i++) {
        int u = -1;
        for(int j=0; j<V; j++)
            if(!inMST[j] && (u==-1 || key[j]<key[u]))
                u = j;

        inMST[u] = true;

        for(int v=0; v<V; v++)
            if(graph[u][v] && !inMST[v] && graph[u][v] < key[v])
                key[v] = graph[u][v], parent[v] = u;
    }

    cout << "MST (Prim's Algorithm):\n";
    cout << "CSE(0) MECH(1) LIB(2) EEE(3) CIVIL(4) CANTEEN(5)\n\n";
    cout << "Edge   Distance\n";
    int total = 0;
    for(int i=1; i<V; i++) {
        cout << parent[i] << " - " << i << "     " << graph[i][parent[i]] << " m\n";
        total += graph[i][parent[i]];
    }
    cout << "Total minimum distance: " << total << " meters\n";

    return 0;
}
