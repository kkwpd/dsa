#include <iostream>
#include <stack>
using namespace std;

struct Node {
    char data;
    Node* left;
    Node* right;
    Node(char val) {
        data = val;
        left = right = NULL;
    }
};

class ExpressionTree {
    Node* root;

public:
    ExpressionTree() {
        root = NULL;
    }

    // Build tree from postfix (manual operand check)
    void constructTree(const string& postfix) {
        stack<Node*> st;

        for (char c : postfix) {
            // Manual check instead of isalnum
            if ((c >= 'a' && c <= 'z') || 
                (c >= 'A' && c <= 'Z') || 
                (c >= '0' && c <= '9')) {
                // Operand → push as leaf
                st.push(new Node(c));
            } else {
                // Operator → pop two nodes and make children
                Node* right = st.top(); st.pop();
                Node* left = st.top(); st.pop();

                Node* node = new Node(c);
                node->left = left;
                node->right = right;

                st.push(node);
            }
        }

        // Root of final expression tree
        root = (!st.empty()) ? st.top() : nullptr;
    }

    // Recursive traversals
    void preorderRec(Node* node) {
        if (!node) return;
        cout << node->data << " ";
        preorderRec(node->left);
        preorderRec(node->right);
    }
    void inorderRec(Node* node) {
        if (!node) return;
        inorderRec(node->left);
        cout << node->data << " ";
        inorderRec(node->right);
    }
    void postorderRec(Node* node) {
        if (!node) return;
        postorderRec(node->left);
        postorderRec(node->right);
        cout << node->data << " ";
    }

    // Iterative traversals
    void preorderIter() {
        if (!root) return;
        stack<Node*> st;
        st.push(root);
        while (!st.empty()) {
            Node* curr = st.top(); st.pop();
            cout << curr->data << " ";
            if (curr->right) st.push(curr->right);
            if (curr->left) st.push(curr->left);
        }
    }
    void inorderIter() {
        stack<Node*> st;
        Node* curr = root;
        while (curr != NULL || !st.empty()) {
            while (curr != NULL) {
                st.push(curr);
                curr = curr->left;
            }
            curr = st.top(); st.pop();
            cout << curr->data << " ";
            curr = curr->right;
        }
    }
    void postorderIter() {
        if (!root) return;
        stack<Node*> st1, st2;
        st1.push(root);
        while (!st1.empty()) {
            Node* curr = st1.top(); st1.pop();
            st2.push(curr);
            if (curr->left) st1.push(curr->left);
            if (curr->right) st1.push(curr->right);
        }
        while (!st2.empty()) {
            cout << st2.top()->data << " ";
            st2.pop();
        }
    }

    // Menu function
    void menu() {
        int choice;
        string postfix;

        do {
            cout << "\n----- Expression Tree Menu -----\n";
            cout << "1. Build Expression Tree (Postfix Input)\n";
            cout << "2. Preorder Traversal (Recursive)\n";
            cout << "3. Inorder Traversal (Recursive)\n";
            cout << "4. Postorder Traversal (Recursive)\n";
            cout << "5. Preorder Traversal (Iterative)\n";
            cout << "6. Inorder Traversal (Iterative)\n";
            cout << "7. Postorder Traversal (Iterative)\n";
            cout << "8. Exit\n";
            cout << "Enter your choice: ";
            cin >> choice;

            switch(choice) {
                case 1:
                    cout << "Enter postfix expression: ";
                    cin >> postfix;
                    constructTree(postfix);
                    cout << "Expression tree built successfully!\n";
                    break;
                case 2:
                    if (root) { cout << "Preorder (Recursive): "; preorderRec(root); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 3:
                    if (root) { cout << "Inorder (Recursive): "; inorderRec(root); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 4:
                    if (root) { cout << "Postorder (Recursive): "; postorderRec(root); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 5:
                    if (root) { cout << "Preorder (Iterative): "; preorderIter(); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 6:
                    if (root) { cout << "Inorder (Iterative): "; inorderIter(); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 7:
                    if (root) { cout << "Postorder (Iterative): "; postorderIter(); cout << endl; }
                    else cout << "Build the tree first!\n";
                    break;
                case 8:
                    cout << "Exiting...\n";
                    break;
                default:
                    cout << "Invalid choice!\n";
            }
        } while(choice != 8);
    }
};

// ---------------- Main ----------------
int main() {
    ExpressionTree et;
    et.menu();
    return 0;
}
