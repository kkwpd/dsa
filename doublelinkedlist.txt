#include <iostream>
using namespace std;

struct node {
    int data;
    node* prev;
    node* next;
};

class DoublyLinkedList {
    node* head;

public:
    DoublyLinkedList() {
        head = NULL;
    }

    // Insert at beginning
    void insert_beg(int x) {
        node* p = new node;
        p->data = x;
        p->prev = NULL;
        p->next = head;

        if (head != NULL)
            head->prev = p;

        head = p;
    }

    // Insert at end
    void insert_end(int x) {
        node* p = new node;
        p->data = x;
        p->next = NULL;

        if (head == NULL) {
            p->prev = NULL;
            head = p;
            return;
        }

        node* ptr = head;
        while (ptr->next != NULL)
            ptr = ptr->next;

        ptr->next = p;
        p->prev = ptr;
    }

    // Insert after value y
    void insert_bet(int x, int y) {
        node* ptr = head;

        // SAFE SEARCH
        while (ptr != NULL && ptr->data != y)
            ptr = ptr->next;

        if (ptr == NULL) {
            cout << "Value " << y << " not found.\n";
            return;
        }

        node* p = new node;
        p->data = x;

        p->next = ptr->next;
        p->prev = ptr;

        if (ptr->next != NULL)
            ptr->next->prev = p;

        ptr->next = p;
    }

    // Delete at beginning
    void delete_beg() {
        if (head == NULL) {
            cout << "List is empty.\n";
            return;
        }

        node* temp = head;
        head = head->next;

        if (head != NULL)
            head->prev = NULL;

        delete temp;
    }

    // Delete at end
    void delete_end() {
        if (head == NULL) {
            cout << "List is empty.\n";
            return;
        }

        if (head->next == NULL) {
            delete head;
            head = NULL;
            return;
        }

        node* ptr = head;

        while (ptr->next != NULL)
            ptr = ptr->next;

        ptr->prev->next = NULL;
        delete ptr;
    }

    // Delete specific value y
    void delete_bet(int y) {
        if (head == NULL) {
            cout << "List is empty.\n";
            return;
        }

        // If value at head
        if (head->data == y) {
            delete_beg();
            return;
        }

        node* ptr = head;

        // SAFE SEARCH
        while (ptr != NULL && ptr->data != y)
            ptr = ptr->next;

        if (ptr == NULL) {
            cout << "Value " << y << " not found.\n";
            return;
        }

        ptr->prev->next = ptr->next;

        if (ptr->next != NULL)
            ptr->next->prev = ptr->prev;

        delete ptr;
    }

    // Display list
    void display() {
        if (head == NULL) {
            cout << "List is empty.\n";
            return;
        }

        node* ptr = head;
        while (ptr != NULL) {
            cout << ptr->data << " ";
            ptr = ptr->next;
        }
        cout << endl;
    }
};

int main() {
    DoublyLinkedList list;
    int choice, x, y;

    do {
        cout << "\n Doubly Linked List Menu \n";
        cout << "1. Insert at Beginning\n";
        cout << "2. Insert at End\n";
        cout << "3. Insert in Between (after value)\n";
        cout << "4. Delete at Beginning\n";
        cout << "5. Delete at End\n";
        cout << "6. Delete specific value\n";
        cout << "7. Display List\n";
        cout << "0. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter value: ";
                cin >> x;
                list.insert_beg(x);
                break;

            case 2:
                cout << "Enter value: ";
                cin >> x;
                list.insert_end(x);
                break;

            case 3:
                cout << "Enter value to insert: ";
                cin >> x;
                cout << "Enter value after which to insert: ";
                cin >> y;
                list.insert_bet(x, y);
                break;

            case 4:
                list.delete_beg();
                break;

            case 5:
                list.delete_end();
                break;

            case 6:
                cout << "Enter value to delete: ";
                cin >> y;
                list.delete_bet(y);
                break;

            case 7:
                list.display();
                break;

            case 0:
                cout << "Exiting...\n";
                break;

            default:
                cout << "Invalid choice.\n";
        }
    } while (choice != 0);

    return 0;
}
