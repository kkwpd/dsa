#include <iostream>
#include <string>
using namespace std;

class Student {
public:
    int roll_no;
    float sgpa;
    string name;

    void accept() {
        cout << "Enter Roll No: ";
        cin >> roll_no;
        cout << "Enter SGPA: ";
        cin >> sgpa;
        cout << "Enter Name: ";
        cin >> name;
    }

    void display() {
        cout << "Roll: " << roll_no << " | SGPA: " << sgpa << " | Name: " << name << endl;
    }

    // Partition for Quick Sort (by SGPA - descending for toppers)
    static int partition(Student s[], int low, int high) {
        float pivot = s[high].sgpa;
        int i = low - 1;

        for (int j = low; j < high; j++) {
            if (s[j].sgpa >= pivot) {  // >= for descending order (toppers first)
                i++;
                swap(s[i], s[j]);
            }
        }
        swap(s[i + 1], s[high]);
        return i + 1;
    }

    // Quick Sort - Descending by SGPA (Highest first)
    static void quickSort(Student s[], int low, int high) {
        if (low < high) {
            int pi = partition(s, low, high);
            quickSort(s, low, pi - 1);
            quickSort(s, pi + 1, high);
        }
    }

    // Bubble Sort by SGPA (ascending)
    static void bubbleSort(Student s[], int n) {
        for (int i = 0; i < n - 1; i++)
            for (int j = 0; j < n - i - 1; j++)
                if (s[j].sgpa > s[j + 1].sgpa)
                    swap(s[j], s[j + 1]);
    }

    // Insertion Sort by Name
    static void insertionSort(Student s[], int n) {
        for (int i = 1; i < n; i++) {
            Student key = s[i];
            int j = i - 1;
            while (j >= 0 && s[j].name > key.name) {
                s[j + 1] = s[j];
                j--;
            }
            s[j + 1] = key;
        }
    }

    // Linear Search by SGPA
    static void linearSearch(Student s[], int n, float key) {
        bool found = false;
        for (int i = 0; i < n; i++) {
            if (s[i].sgpa == key) {
                s[i].display();
                found = true;
            }
        }
        if (!found) cout << "No student with SGPA " << key << " found.\n";
    }

    // Binary Search by Name (array must be sorted by name)
    static void binarySearch(Student s[], int n, string key) {
        int l = 0, r = n - 1;
        while (l <= r) {
            int mid = (l + r) / 2;
            if (s[mid].name == key) {
                s[mid].display();
                return;
            }
            if (s[mid].name < key)
                l = mid + 1;
            else
                r = mid - 1;
        }
        cout << "Student '" << key << "' not found.\n";
    }
};

int main() {
    int n;
    cout << "Enter number of students: ";
    cin >> n;

    Student s[100];

    cout << "\n--- Enter Student Details ---\n";
    for (int i = 0; i < n; i++) {
        cout << "\nStudent " << i + 1 << ":\n";
        s[i].accept();
    }

    int choice;
    do {
        cout << "\n=== MENU ===\n";
        cout << "1. Quick Sort - Top 10 Toppers (by SGPA)\n";
        cout << "2. Bubble Sort by SGPA (ascending)\n";
        cout << "3. Insertion Sort by Name\n";
        cout << "4. Linear Search by SGPA\n";
        cout << "5. Binary Search by Name\n";
        cout << "6. Display All Students\n";
        cout << "0. Exit\n";
        cout << "Choose: ";
        cin >> choice;

        switch (choice) {
            case 1:
                Student::quickSort(s, 0, n - 1);
                cout << "\nTop 10 Toppers (Highest SGPA First):\n";
                cout << "------------------------------------------------\n";
                for (int i = 0; i < n && i < 10; i++) {
                    cout << i + 1 << ". ";
                    s[i].display();
                }
                break;

            case 2:
                Student::bubbleSort(s, n);
                cout << "\nSorted by SGPA (Low to High):\n";
                for (int i = 0; i < n; i++) s[i].display();
                break;

            case 3:
                Student::insertionSort(s, n);
                cout << "\nSorted by Name:\n";
                for (int i = 0; i < n; i++) s[i].display();
                break;

            case 4: {
                float sgpa;
                cout << "Enter SGPA to search: ";
                cin >> sgpa;
                cout << "Students with SGPA " << sgpa << ":\n";
                Student::linearSearch(s, n, sgpa);
                break;
            }

            case 5: {
                string name;
                cout << "Enter Name to search: ";
                cin >> name;
                Student::insertionSort(s, n);  // Must sort by name first
                Student::binarySearch(s, n, name);
                break;
            }

            case 6:
                cout << "\nAll Students:\n";
                for (int i = 0; i < n; i++) s[i].display();
                break;

            case 0:
                cout << "Good luck for your exam! You got this!\n";
                break;

            default:
                cout << "Invalid choice!\n";
        }
    } while (choice != 0);

    return 0;
}
