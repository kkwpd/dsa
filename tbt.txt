#include <iostream>
using namespace std;

class node {
public:
    int data;
    node* left;
    node* right;
    bool lthread, rthread;

    node(int data) {
        this->data = data;
        this->left = nullptr;
        this->right = nullptr;
        this->lthread = true;  // True means pointer is a thread
        this->rthread = true;
    }
};

class ThreadedBinaryTree {
public:
    node* root;

    ThreadedBinaryTree() {
        root = nullptr;
    }

    node* insert(node* root, int data) {
        node* curr = root;
        node* par = nullptr;

        // Find the parent node under which new node will be inserted
        while (curr != nullptr) {
            if (data == curr->data) {
                cout << "Duplicate value: " << data << endl;
                return root;
            }
            par = curr;
            if (data < curr->data) {
                if (!curr->lthread)
                    curr = curr->left;
                else
                    break;
            } else {
                if (!curr->rthread)
                    curr = curr->right;
                else
                    break;
            }
        }

        node* temp = new node(data);

        if (par == nullptr) {
            // Tree is empty
            root = temp;
        } else if (data < par->data) {
            temp->left = par->left;
            temp->right = par;
            par->lthread = false;
            par->left = temp;
        } else {
            temp->left = par;
            temp->right = par->right;
            par->rthread = false;
            par->right = temp;
        }
        return root;
    }

    // Helper function to find the leftmost node of the tree
    node* leftmost(node* root) {
        if (root == nullptr)
            return nullptr;
        while (!root->lthread)
            root = root->left;
        return root;
    }

    void inorder() {
        if (root == nullptr)
            return;

        node* curr = leftmost(root);

        while (curr != nullptr) {
            cout << curr->data << " ";

            if (curr->rthread)
                curr = curr->right;
            else
                curr = leftmost(curr->right);
        }
    }

    // Wrapper for insert
    void insert(int data) {
        root = insert(root, data);
    }
};

int main() {
    ThreadedBinaryTree tbt;

    tbt.insert(20);
    tbt.insert(30);
    tbt.insert(9);
    tbt.insert(8);
    tbt.insert(45);
    tbt.insert(12);

    cout << "In-order Traversal: ";
    tbt.inorder();
    cout << endl;

    return 0;
}
