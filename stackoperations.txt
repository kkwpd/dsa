#include <iostream>
#include <cctype>
#include <cmath>  // for pow()
using namespace std;

struct Node {
    char data;
    Node* next;
};

// Stack ADT using singly linked list
class Stack {
    Node* top;

public:
    char arr[100];
    Stack() { top = NULL; }

    bool isEmpty() { return top == NULL; }

    void push(char x) {
        Node* p = new Node();
        p->data = x;
        p->next = top;
        top = p;
    }

    char pop() {
        if (top == NULL) return '\0';
        Node* temp = top;
        char val = temp->data;
        top = top->next;
        delete temp;
        return val;
    }

    char peek() {
        if (top == NULL) return '\0';
        return top->data;
    }

    void input() {
        cin.ignore();
        cout << "Enter expression: ";
        cin.getline(arr, 100);
    }

    bool isOperand(char ch) {
        return isdigit(ch);
    }

    bool isOperator(char ch) {
        return (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '^');
    }

    int precedence(char op) {
        if (op == '+' || op == '-') return 1;
        if (op == '*' || op == '/') return 2;
        if (op == '^') return 3;
        return 0;
    }

    // Infix to Postfix conversion
    void convert() {
        Stack s;  // use another stack for operators
        for (int i = 0; arr[i] != '\0'; ++i) {
            char ch = arr[i];

            if (ch == ' ') continue;

            if (isOperand(ch)) {
                cout << ch << " ";
            } else if (ch == '(') {
                s.push(ch);
            } else if (ch == ')') {
                while (!s.isEmpty() && s.peek() != '(') {
                    cout << s.pop() << " ";
                }
                if (s.peek() == '(') s.pop();
            } else if (isOperator(ch)) {
                while (!s.isEmpty() && isOperator(s.peek()) &&
                       ((precedence(s.peek()) > precedence(ch)) ||
                        (precedence(s.peek()) == precedence(ch) && ch != '^'))) {
                    cout << s.pop() << " ";
                }
                s.push(ch);
            }
        }
        while (!s.isEmpty()) {
            char c = s.pop();
            if (c != '(' && c != ')')
                cout << c << " ";
        }
        cout << endl;
    }

    // Postfix evaluation
    void evaluate() {
        Stack s;  // use stack for operands
        for (int i = 0; arr[i] != '\0'; ++i) {
            char ch = arr[i];

            if (ch == ' ') continue;

            if (isOperand(ch)) {
                s.push(ch - '0'); // convert char → int
            } else if (isOperator(ch)) {
                int val2 = s.pop();
                int val1 = s.pop();
                int result = 0;

                switch (ch) {
                    case '+': result = val1 + val2; break;
                    case '-': result = val1 - val2; break;
                    case '*': result = val1 * val2; break;
                    case '/': result = val1 / val2; break;
                    case '^': result = pow(val1, val2); break;
                }
                s.push(result);
            }
        }
        cout << "Result = " << (int)s.pop() << endl;
    }
};

int main() {
    Stack s;
    int choice;

    while (true) {
        cout << "\nMenu:\n";
        cout << "1. convert infix to postfix\n";
        cout << "2. evaluate postfix \n";
        cout << "3. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                s.input();
                s.convert();
                break;
            case 2:
                s.input();
                s.evaluate();
                break;
            case 3:
                cout << "Exiting the program.\n";
                return 0;
            default:
                cout << "Invalid choice! Please try again.\n";
        }
    }
    return 0;
}
